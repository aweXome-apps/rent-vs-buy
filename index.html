    <script>

        function calculate() {

            // Retrieve and parse inputs

            const S = parseFloat(document.getElementById('initialSavings').value);           // Initial savings

            const I = parseFloat(document.getElementById('annualIncome').value);            // Annual income

            const E = parseFloat(document.getElementById('annualExpenses').value);          // Annual non-housing expenses

            const r = parseFloat(document.getElementById('investmentRate').value) / 100;    // Investment return rate

            const P = parseFloat(document.getElementById('housePrice').value);              // House price

            const D = parseFloat(document.getElementById('downPayment').value);             // Down payment

            const mortgageRate = parseFloat(document.getElementById('mortgageRate').value) / 100; // Mortgage interest rate

            const propertyTaxRate = parseFloat(document.getElementById('propertyTaxRate').value) / 100; // Property tax rate

            const insuranceCost = parseFloat(document.getElementById('insuranceCost').value); // Insurance cost

            const maintenanceCost = parseFloat(document.getElementById('maintenanceCost').value); // Maintenance cost

            const a = parseFloat(document.getElementById('appreciationRate').value) / 100;  // House appreciation rate

            const R = parseFloat(document.getElementById('monthlyRent').value);             // Monthly rent

            const rentIncreaseRate = parseFloat(document.getElementById('rentIncreaseRate').value) / 100; // Rent increase rate


            // Calculate monthly mortgage payment

            const loanAmount = P - D;

            const monthlyMortgageRate = mortgageRate / 12;

            const numberOfPayments = 30 * 12; // 30-year mortgage

            const M_buy = loanAmount * (monthlyMortgageRate * Math.pow(1 + monthlyMortgageRate, numberOfPayments)) / 

                          (Math.pow(1 + monthlyMortgageRate, numberOfPayments) - 1);


            // Arrays for charting

            const years = Array.from({length: 31}, (_, i) => i); // Years 0 to 30

            let netWorthBuy = [];

            let netWorthRent = [];


            // Buying scenario

            let liquidAssetsBuy = S - D; // Initial liquid assets after down payment

            let houseValue = P;

            let mortgageBalance = loanAmount;

            let equity = houseValue - mortgageBalance;

            netWorthBuy.push(liquidAssetsBuy + equity); // Net worth at year 0


            for (let t = 1; t <= 30; t++) {

                const propertyTax = propertyTaxRate * houseValue;

                const annualHousingCost = 12 * M_buy + propertyTax + insuranceCost + maintenanceCost;

                liquidAssetsBuy = liquidAssetsBuy * (1 + r) + I - E - annualHousingCost;


                houseValue *= (1 + a); // House appreciates


                // Simplified mortgage balance update (annual approximation)

                const interestPaid = mortgageBalance * mortgageRate;

                const principalPaid = (12 * M_buy) - interestPaid;

                mortgageBalance -= principalPaid;

                if (mortgageBalance < 0) mortgageBalance = 0;


                equity = houseValue - mortgageBalance;

                netWorthBuy.push(liquidAssetsBuy + equity);

            }


            // Renting scenario

            let liquidAssetsRent = S; // Keep all savings initially

            let currentRent = R;

            netWorthRent.push(liquidAssetsRent); // Net worth at year 0


            for (let t = 1; t <= 30; t++) {

                const annualRentCost = 12 * currentRent;

                liquidAssetsRent = liquidAssetsRent * (1 + r) + I - E - annualRentCost;

                netWorthRent.push(liquidAssetsRent);

                currentRent *= (1 + rentIncreaseRate); // Rent increases annually

            }


            // Render chart

            const ctx = document.getElementById('netWorthChart').getContext('2d');

            if (window.myChart) window.myChart.destroy(); // Clear previous chart if exists

            window.myChart = new Chart(ctx, {

                type: 'line',

                data: {

                    labels: years,

                    datasets: [

                        { label: 'Buying', data: netWorthBuy, borderColor: 'blue', fill: false },

                        { label: 'Renting', data: netWorthRent, borderColor: 'green', fill: false }

                    ]

                },

                options: {

                    responsive: true,

                    scales: {

                        x: { title: { display: true, text: 'Years' } },

                        y: { title: { display: true, text: 'Net Worth ($)' } }

                    }

                }

            });

        }

    </script>
